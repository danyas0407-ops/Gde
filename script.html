let peer = null; let hostConnection = null; // Для гостей (связь с хостом) let guestConnections = []; // Для хоста (список гостей) let players = []; // Список всех игроков let isHost = false; let myName = ""; // Вспомогательная функция: переключает интерфейс function showGameScreen(code) { document.getElementById('lobby-screen').classList.add('hidden'); document.getElementById('game-screen').classList.remove('hidden'); document.getElementById('roomTitle').innerText = `Комната: ${code}`; if (isHost) { document.getElementById('startBtn').classList.remove('hidden'); } } // Обновление списка игроков на экране function updatePlayersUI() { const ul = document.getElementById('playersUl'); ul.innerHTML = ''; players.forEach(p => { const li = document.createElement('li'); li.innerText = p.name; ul.appendChild(li); }); } // ------------------------------------------------ // ЛОГИКА СОЗДАТЕЛЯ КОМНАТЫ (ХОСТА) // ------------------------------------------------ function createRoom() { myName = document.getElementById('playerName').value.trim(); if (!myName) return alert('Введи имя!'); // Генерируем 4 буквы кода const code = Math.random().toString(36).substring(2, 6).toUpperCase(); const fullRoomId = 'chepuha-game-' + code; // Уникальный ID комнаты // Создаем сервер прямо в браузере! peer = new Peer(fullRoomId); peer.on('open', (id) => { isHost = true; players.push({ name: myName, id: id }); updatePlayersUI(); showGameScreen(code); }); // Когда кто-то подключается по коду peer.on('connection', (conn) => { guestConnections.push(conn); // Слушаем сообщения от гостей conn.on('data', (data) => { if (data.type === 'join') { players.push({ name: data.name, id: conn.peer }); updatePlayersUI(); // Рассылаем обновленный список всем гостям guestConnections.forEach(c => c.send({ type: 'updatePlayers', players: players })); } }); }); peer.on('error', (err) => { alert("Ошибка: " + err.type); }); } // ------------------------------------------------ // ЛОГИКА ГОСТЯ (ПРИСОЕДИНИТЬСЯ) // ------------------------------------------------ function joinRoom() { myName = document.getElementById('playerName').value.trim(); const code = document.getElementById('roomCode').value.trim().toUpperCase(); if (!myName || !code) return alert('Введи имя и код!'); const fullRoomId = 'chepuha-game-' + code; // Создаем клиента peer = new Peer(); peer.on('open', (id) => { // Подключаемся к хосту по коду hostConnection = peer.connect(fullRoomId); hostConnection.on('open', () => { // Говорим хосту своё имя hostConnection.send({ type: 'join', name: myName }); showGameScreen(code); }); // Слушаем команды от хоста hostConnection.on('data', (data) => { if (data.type === 'updatePlayers') { players = data.players; updatePlayersUI(); } }); }); peer.on('error', (err) => { alert("Не удалось подключиться. Проверь код!"); }); } // Заглушка кнопки старта function startGame() { alert("Игра началась! (Логику вопросов добавим дальше)"); }
